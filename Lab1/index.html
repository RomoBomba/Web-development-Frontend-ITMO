Для программистов

Пример кода для генерации аккорда


function generateChord(root, type) {
    const chords = {
        'C-major': ['C', 'E', 'G'],
        'G-major': ['G', 'B', 'D'],
        'A-minor': ['A', 'C', 'E']
    };

    return chords[`${root}-${type}`] || ['Неизвестный аккорд'];
}

console.log(generateChord('C', 'major')); // ['C', 'E', 'G']
