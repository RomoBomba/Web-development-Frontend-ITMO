Формула расчета аннуитетного платежа

ЕП = S × (P + P ÷ ((1 + P)N - 1))
ЕП — ежемесячный платеж
S — сумма кредита
P — процентная ставка (месячная)
N — срок кредита (в месяцах)

Для разработчиков

Генератор музыкальных аккордов


// Функция для генерации аккордов
function generateChord(root, type) {
    const chords = {
        'C-major': ['C', 'E', 'G'],
        'G-major': ['G', 'B', 'D'],
        'A-minor': ['A', 'C', 'E']
    };
    return chords[`${root}-${type}`] || ['Аккорд не найден'];
}

// Пример использования
console.log(generateChord('C', 'major'));
